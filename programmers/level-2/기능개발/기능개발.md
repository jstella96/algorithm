## 풀이 코드

```java
//길이가 정해지지 않은 배열이 답일 때 답은 보통 어디에 저장하시나요?
//-> 리스트 이당.
import java.util.*;

class Work{
	int progress;
	double speed;
	public Work(int progress, int speed){
		this.progress=progress;
		this.speed=speed;
	}
}

class Solution {
    public int[] solution(int[] progresses, int[] speeds) {

        Queue<Integer> answerQueue = new LinkedList<Integer>();
        Queue<Work> queue = new LinkedList<Work>();
        for(int i=0 ;  i < progresses.length ; i++){
            queue.offer(new Work(progresses[i], speeds[i]));
        }
        while(!queue.isEmpty()){
            Work tmp = queue.poll();
            double tmpTimeRequired = Math.ceil( (100 - tmp.progress) / tmp.speed);
            int num = 1;
             while(!queue.isEmpty()){
                double  workTimeRequired = Math.ceil( (100 - queue.peek().progress) / queue.peek().speed);
                if(workTimeRequired <= tmpTimeRequired){
                    queue.poll();
                    num++;
                }else{
                    break;
                }
            }
            answerQueue.offer(num);
        }

        //답 배열로 이동
        int answerLen = answerQueue.size();
        int[]  answer = new int[answerLen];
        for(int z = 0;z < answerLen; z++){
            answer[z] = answerQueue.poll();
        }

        return answer;

    }
}
```
